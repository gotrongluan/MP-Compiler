
            function countOdd(x:array[0 .. 4] of integer):integer;
            var i,count:integer;
            begin
                count := 0;
                for i:= 0 to 4 do
                    if (x[i] mod 2) <> 0 then
                        count := count + 1;
                return count; 
            end
