
            function counting(n: integer): integer;
            var i, j, dup, count: integer;
            begin
                dup := count := 0;
                i := 0;
                while i <= 10 do
                begin
                    j := 0;
                    while j <= 10 do
                    begin
                        if (arr[i] + arr[j]) = n then
                        begin
                            count := count + 1;
                            if i = j then
                                dup := dup + 1;
                        end
                        j := j + 1;
                    end
                    i := i + 1;
                end
                return ((count - dup) div 2 + dup);
            end
