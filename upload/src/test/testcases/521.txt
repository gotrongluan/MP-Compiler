
            function counting(n: integer): integer;
            var i, j, dup, count: integer;
            begin
                dup := count := 0;
                for i := 0 to 10 do
                begin
                    for j := 0 to 10 do
                    begin
                        if (arr[i] + arr[j]) = n then
                        begin
                            count := count + 1;
                            if i = j then
                                dup := dup + 1;
                        end
                    end
                end
                return ((count - dup) div 2 + dup);
            end
